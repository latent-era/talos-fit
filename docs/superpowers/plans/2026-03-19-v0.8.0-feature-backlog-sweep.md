- [ ] **Step 4: Implement MotionStartDetector**

```kotlin
sealed class MotionStartEvent {
    data class CountdownTick(val remainingMs: Long) : MotionStartEvent()
    object Started : MotionStartEvent()
    object Cancelled : MotionStartEvent()
}

class MotionStartDetector(
    private val loadThresholdKg: Float = 5f,
    private val holdDurationMs: Long = 1500L
) {
    private val _events = MutableSharedFlow<MotionStartEvent>(replay = 1)
    val events: SharedFlow<MotionStartEvent> = _events

    private var holdStartMs: Long? = null

    suspend fun onMetricReceived(load: Float, timestampMs: Long) {
        if (load >= loadThresholdKg) {
            val start = holdStartMs ?: run {
                holdStartMs = timestampMs
                timestampMs
            }
            val elapsed = timestampMs - start
            if (elapsed >= holdDurationMs) {
                _events.emit(MotionStartEvent.Started)
                holdStartMs = null
            } else {
                _events.emit(MotionStartEvent.CountdownTick(holdDurationMs - elapsed))
            }
        } else {
            if (holdStartMs != null) {
                _events.emit(MotionStartEvent.Cancelled)
                holdStartMs = null
            }
        }
    }

    fun reset() { holdStartMs = null }
}
```

- [ ] **Step 1: Implement SafeWordDetectionManager**

Manages lifecycle: starts listening when workout begins, stops when workout ends or app backgrounds. Auto-restarts if recognition stops unexpectedly (within 1 second). Holds reference to `SafeWordListener`.

```kotlin
class SafeWordDetectionManager(
    private val preferencesManager: PreferencesManager
) {
    private var listener: SafeWordListener? = null

    fun startForWorkout() {
        val safeWord = preferencesManager.safeWord ?: return
        if (!preferencesManager.voiceStopEnabled) return
        listener = SafeWordListener(safeWord).also { it.startListening() }
    }

    fun stop() {
        listener?.stopListening()
        listener = null
    }

    val detectedWord: Flow<String>
        get() = listener?.detectedWord ?: emptyFlow()
}
```
